Create Table Customer_Info(
	
	Customer_ID SERIAL                          PRIMARY KEY ,
	First_Name VARCHAR(20)                      NOT NULL    ,
	Last_Name VARCHAR(20)                                   ,
	Mobile_Number BIGINT                        NOT NULL
	
);

Create Table Customer_Auth(
	
	Customer_ID INT                                     ,
	E_Mail_ID VARCHAR(50)                       UNIQUE  ,
	Password BIGINT                             NOT NULL,
	
	FOREIGN KEY (Customer_ID) REFERENCES Customer_Info(Customer_ID)
);

CREATE TABLE Order_Details(
	
	Order_ID SERIAL                             PRIMARY KEY,
	Customer_ID INT                                        ,
	Parcel_Type VARCHAR(20)                                , 
	Parcel_Weight_KG NUMERIC(5,2)                          ,
	EST_Distance REAL                                    ,
	Order_Placed_Date TIMESTAMP                            ,
	EST_Delivery_Date TIMESTAMP                            ,
	Dimensions VARCHAR(30)                                 ,
	Fee NUMERIC(7,2)                                       ,
	Payment_Type VARCHAR(10)                               ,
	
	FOREIGN KEY (Customer_ID) REFERENCES Customer_Info(Customer_ID)
);

Create Table Sender_Details(
	
	Order_ID INT                                           ,
	Customer_ID INT                                        ,
	Name VARCHAR(30)                                     
	Pincode INT                                 NOT NULL,
	City VARCHAR(20)                            NOT NULL,
	Sender_Address VARCHAR(100)                 NOT NULL,
	Contact_Number BIGINT                               ,
	
	FOREIGN KEY (Customer_ID) REFERENCES Customer_Info(Customer_ID),
	FOREIGN KEY (Order_ID) REFERENCES Order_Details(Order_ID)
	
);

Create Table Recipient_Details(
	
	Order_ID INT                                           ,
	Customer_ID INT                                        , 
	Name VARCHAR(50)                               NOT NULL,
	Pincode INT                                    NOT NULL,                                       
	City VARCHAR(20)                                       ,
	Delivery_Address VARCHAR(100)                  NOT NULL,
	Contact_Number BIGINT                          NOT NULL,      
	
	FOREIGN KEY (Customer_ID) REFERENCES Customer_Info(Customer_ID),
	FOREIGN KEY (Order_ID) REFERENCES Order_Details(Order_ID)
	
);
